15 - (void) drawStimulusInRect: (CGRect) dirtyRect { //console.info("FractController>drawStimulusInRect");
16  var temp = [Misc logMARfromDecVA: [Misc decVAFromGapPixels: stimStrengthInDeviceunits]];
17  [trialHistoryController setValue: [Misc stringFromNumber: temp decimals: 3 localised: NO]];
18  if ([Settings crowdingType] > 0) {
19  if (currentTestName != "Acuity_Vernier") { // don't do crowding with Vernier etc.
20  CGContextSaveGState(cgc);
21  CGContextTranslateCTM(cgc, viewWidth / 2, viewHeight / 2); // origin to center
22  CGContextTranslateCTM(cgc, -xEcc, -yEcc);
23  var i, crowdingDistance = [self acuityCrowdingDistanceFromGap: stimStrengthInDeviceunits];
24  switch ([Settings crowdingType]) {
25  case 0: break; // should not occur here
26  case 1: // flanking bars
27  var distance2 = 1.5 * crowdingDistance / 2;
28  var length2 = stimStrengthInDeviceunits * 2.5;
29  CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
30  [optotypes strokeVLineAtX: -distance2 y0: -length2 y1: length2];
31  [optotypes strokeVLineAtX: distance2 y0: -length2 y1: length2];
32  break;
33  case 2: // flanking rings
34  for (i = -1; i <= 1; i++) { //console.info(i);
35  var tempX = i * crowdingDistance;
36  CGContextTranslateCTM(cgc, -tempX, 0);
37  if (i != 0) [optotypes drawLandoltWithGapInPx: stimStrengthInDeviceunits landoltDirection: -1];
38  CGContextTranslateCTM(cgc, +tempX, 0);
39  } break;
40  case 3: // surounding bars
41  var distance2 = 1.5 * crowdingDistance / 2;
42  var length2 = stimStrengthInDeviceunits * 4;
43  CGContextSetLineCap(cgc, kCGLineCapRound);
44  CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
45  [optotypes strokeVLineAtX: -distance2 y0: -length2 y1: length2];
46  [optotypes strokeVLineAtX: distance2 y0: -length2 y1: length2];
47  [optotypes strokeHLineAtX0: -length2 y: -distance2 x1: length2];
48  [optotypes strokeHLineAtX0: -length2 y: distance2 x1: length2];
49  break;
50  case 4: // surounding ring
51  CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
52  [optotypes strokeCircleAtX: 0 y: 0 radius: 1.5 * crowdingDistance / 2];
53  break;
54  case 5: // surrunding square
55  var frameSize = 1.5 * crowdingDistance, frameSize2 = frameSize / 2;
56  CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
57  CGContextStrokeRect(cgc, CGRectMake(-frameSize2, -frameSize2, frameSize, frameSize));
58  break;
59  case 6: // row of optotypes
60  for (i = -2; i <= 2; i++) {
61  var directionPresentedX = [Misc iRandom: nAlternatives];
62  var tempX = i * crowdingDistance;
63  CGContextTranslateCTM(cgc, -tempX, 0);
64  if (i != 0) [optotypes drawLandoltWithGapInPx: stimStrengthInDeviceunits landoltDirection: directionPresentedX];
65  CGContextTranslateCTM(cgc, +tempX, 0);
66  } break;
67  }
68  CGContextRestoreGState(cgc);
69  }
70  }
71  [super drawStimulusInRect: dirtyRect];
72 }
