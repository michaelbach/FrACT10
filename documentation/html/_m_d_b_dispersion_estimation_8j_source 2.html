59function findMaxLlhInRange(df, r1, r2, delta) {
60 let lMax = -1, lMarMax, lMar = r1;
61 while (lMar < r2) {
62 const ll = likelihoodFunc(lMar, df); //console.info(lMar, ll);
63 if (ll > lMax) {lMax = ll; lMarMax = lMar;}
64 lMar += delta;
65 }
66 return lMarMax;

71function threshEstimate(df) { // console.info("threshEstimate");
72 let delta = 0.5; // initial LogMAR precision for rough homing-in
73 let lMarMax = findMaxLlhInRange(df, kBestLogMAR, kWorstLogMAR, delta);
74 lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to ±0.1 LogMAR
75 delta /= 10;
76 lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to ±0.02 LogMAR
77 delta /= 10;
78 lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to ±0.004 LogMAR. Overkill??
79 return lMarMax;
80}

128function logisticFun(guessingProbability, inflectionPoint, x) {
129 //console.log("guessingProbability: ", guessingProbability, ", inflectionPoint: ", inflectionPoint);
130 x = 1 - x; inflectionPoint = 1 - inflectionPoint;
131 // 2023-02-07 previously, slope was defined inversely. No change in result, now more readable
132 return guessingProbability + (1 - guessingProbability) / (1 + Math.exp(-gSlopeCI95 * (x - inflectionPoint)));
