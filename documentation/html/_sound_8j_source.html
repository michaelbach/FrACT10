61 - (void) playSoundFromBuffer: (id) buffer { //console.info("Sound>playSoundFromBuffer");
62  if (buffer == nil) return;
63  var source = audioContext.createBufferSource();
64  source.buffer = buffer;
65  source.connect(volumeNode);
66  volumeNode.gain.value = Math.pow([Settings soundVolume] / 100.0, 2); // a more physiologic transfer function IMHO
67  source.start(0);
68 }
