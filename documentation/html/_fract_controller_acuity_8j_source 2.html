28- (void) drawStimulusInRect: (CGRect) dirtyRect { //console.info("FractController>drawStimulusInRect");
29 [trialHistoryController setValue: [MiscSpace logMARfromDecVA: [MiscSpace decVAFromGapPixels: stimStrengthInDeviceunits]]];
30 if ([Settings crowdingType] > 0) {
31 if (currentTestID != kTestAcuityVernier) { // don't do crowding with Vernier etc.
32 CGContextSaveGState(cgc);
33 CGContextTranslateCTM(cgc, viewWidth / 2, viewHeight / 2); // origin to center
34 CGContextTranslateCTM(cgc, -xEccInPix, -yEccInPix);
35 const crowdingDistance = [self acuityCrowdingDistanceFromGap: stimStrengthInDeviceunits];
36 switch ([Settings crowdingType]) {
37 case 0: break; // should not occur here
38 case 1: // flanking bars
39 const distance2 = 1.5 * crowdingDistance / 2;
40 const length2 = stimStrengthInDeviceunits * 2.5;
41 CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
42 [optotypes strokeVLineAtX: -distance2 y0: -length2 y1: length2];
43 [optotypes strokeVLineAtX: distance2 y0: -length2 y1: length2];
44 break;
45 case 2: // flanking rings
46 for (let i = -1; i <= 1; i++) { //console.info(i);
47 const tempX = i * crowdingDistance;
48 CGContextTranslateCTM(cgc, -tempX, 0);
49 if (i != 0) [optotypes drawLandoltWithGapInPx: stimStrengthInDeviceunits landoltDirection: -1];
50 CGContextTranslateCTM(cgc, +tempX, 0);
51 } break;
52 case 3: // surounding bars
53 const distance4 = 1.5 * crowdingDistance / 2;
54 const length4 = stimStrengthInDeviceunits * 4;
55 CGContextSetLineCap(cgc, kCGLineCapRound);
56 CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
57 [optotypes strokeVLineAtX: -distance4 y0: -length4 y1: length4];
58 [optotypes strokeVLineAtX: distance4 y0: -length4 y1: length4];
59 [optotypes strokeHLineAtX0: -length4 y: -distance4 x1: length4];
60 [optotypes strokeHLineAtX0: -length4 y: distance4 x1: length4];
61 break;
62 case 4: // surounding ring
63 CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
64 [optotypes strokeCircleAtX: 0 y: 0 radius: 1.5 * crowdingDistance / 2];
65 break;
66 case 5: // surrunding square
67 const frameSize = 1.5 * crowdingDistance, frameSize2 = frameSize / 2;
68 CGContextSetLineWidth(cgc, stimStrengthInDeviceunits);
69 CGContextStrokeRect(cgc, CGRectMake(-frameSize2, -frameSize2, frameSize, frameSize));
70 break;
71 case 6: // row of optotypes
72 for (i = -2; i <= 2; i++) {
73 const directionPresentedX = [Misc iRandom: nAlternatives];
74 const tempX = i * crowdingDistance;
75 CGContextTranslateCTM(cgc, -tempX, 0);
76 if (i != 0) [optotypes drawLandoltWithGapInPx: stimStrengthInDeviceunits landoltDirection: directionPresentedX];
77 CGContextTranslateCTM(cgc, +tempX, 0);
78 } break;
79 }
80 CGContextRestoreGState(cgc);
81 }
82 }
83 [self drawCenterFixMark];
84 [super drawStimulusInRect: dirtyRect];
85}

234- (float) acuityCrowdingDistanceFromGap: (float) gap {
235 let returnVal = 5 * gap + 2 * gap; // case 0
236 switch ([Settings crowdingDistanceCalculationType]) {
237 case 1:
