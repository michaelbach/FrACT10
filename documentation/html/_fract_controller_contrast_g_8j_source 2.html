26- (void) gratingSineWithPeriodInPx: (float) periodInPx direction: (int) theDirection {
27 //console.info("optotypes>gratingSineWithPeriodInPx: ", periodInPx, theDirection);
28 const l2 = 2 * Math.round(0.5 * 1.42 * Math.max(viewWidth2, viewHeight2));
29 const trigFactor = 1.0 / periodInPx * 2 * Math.PI; // calculate only once
30 CGContextRotateCTM(cgc, -theDirection * 22.5 * Math.PI / 180); // rotate to desired angle
31 CGContextSetLineWidth(cgc, 1.3);
32 let l, lError = 0, lDiscrete;
33 for (let ix = -l2; ix <= l2; ++ix) {
34 l = 0.5 + 0.5 * contrastMichelsonPercent / 100 * Math.sin((ix % periodInPx) * trigFactor);
35 l = [MiscLight devicegrayFromLuminance: l]; // apply gamma correction
36 lDiscrete = l;
37 if ([Settings gratingUseErrorDiffusion]) {
38 l = lError + 255 * l; // map to 0â¦255 and apply previous residual
39 lDiscrete = Math.round(l); // discrete integer values 0â¦255
40 lError = l - lDiscrete; // keep residual (what was lost by rounding) for next time
41 lDiscrete = lDiscrete / 255; // remap to 0â¦1
42 }
43 CGContextSetStrokeColor(cgc, [CPColor colorWithWhite: lDiscrete alpha: 1]);
44 CGContextBeginPath(cgc);
45 CGContextMoveToPoint(cgc, ix, -l2); CGContextAddLineToPoint(cgc, ix, l2);
46 CGContextStrokePath(cgc);
47 }
48 const r = 0.5 * [MiscSpace pixelFromDegree: [Settings gratingDiaInDeg]];
49 const w = r / 20;
50 l = [MiscLight devicegrayFromLuminance: 0.5];
51 [self annulusWithRadius: r - 2 * w width: w gray: l alpha: 0.125];
52 [self annulusWithRadius: r - w width: w gray: l alpha: 0.25];
53 [self annulusWithRadius: r width: w gray: l alpha: 0.5];
54 [self annulusWithRadius: r + w width: w gray: l alpha: 0.75];
