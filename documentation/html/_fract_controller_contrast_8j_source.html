96 // contrast: 0.1 â¦ 100, thresholder: 0 â¦ 1

99 - (float) stimDeviceunitsFromThresholderunits: (float) thresholderunit { //console.info("FractControllerAcuityC>stimDeviceunitsFromThresholderunits");
