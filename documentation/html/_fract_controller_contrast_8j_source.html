96/*
97basic flow:
98 Thresholder â thresholderunits
99 thresholderunits â deviceunits (logCSWeber)
100 deviceunits â upper/lower luminance, considering gamma â discrete values for [CPColor colorWithWhite] (0â¦255)
101 present stimulus
102 stimStrengthInDeviceunits â thresholderunits (currently ignoring gamma because "locally everything is linear")
103 thresholderunits + correctness â Thresholder
104
105
106// contrast: 0.1 â¦ 100, thresholder: 0 â¦ 1
107// deviceUnits are in logCSWeber for all contrast tests; for gratings that is converted to Michelson%
108// logCSW: 2 â¦ 0, thresholder: 0 â¦ 1 */
109- (float) stimDeviceunitsFromThresholderunits: (float) thresholderunit { //console.info("FractControllerAcuityC>stimDeviceunitsFromThresholderunits");
110 const logCSWMaximal = [Settings contrastMaxLogCSWeber];
111 const deviceVal = logCSWMaximal - logCSWMaximal * thresholderunit; // logCSWMinimal = 0 anyway
112 return deviceVal;
113}
114- (float) stimThresholderunitsFromDeviceunits: (float) d { //console.info("FractControllerAcuityC>stimThresholderunitsFromDeviceunits");
115 //console.info("d: ", d, ", retVal: ", retVal)
116 const logCSWMaximal = [Settings contrastMaxLogCSWeber];
117 const retVal = (logCSWMaximal - d) / logCSWMaximal
118 return retVal;
119}
