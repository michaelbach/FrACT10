51 function findMaxLlhInRange(df, r1, r2, delta) {
52  var lMax = -1, lMarMax, lMar = r1;
53  while (lMar < r2) {
54  var ll = likelihoodFunc(lMar, df); //console.info(lMar, ll);

60 function threshEstimate(df) { // console.info("threshEstimate");
61  var delta = 0.5; // initial LogMAR precision for rough homing-in
62  var lMarMax = findMaxLlhInRange(df, kBestLogMAR, kWorstLogMAR, delta);
63  lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to Â±0.1 LogMAR
64  delta /= 10;
65  lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to Â±0.02 LogMAR
66  delta /= 10;
67  lMarMax = findMaxLlhInRange(df, lMarMax - delta, lMarMax + delta, delta / 5); // now precise to Â±0.004 LogMAR. Overkill??
68  return lMarMax;

111 function logisticFun(guessingProbability, inflectionPoint, x) {
112  //console.log("guessingProbability: ", guessingProbability, ", inflectionPoint: ", inflectionPoint);
113  var slope = 0.1;
114  x = 1 - x; inflectionPoint = 1 - inflectionPoint;
115  return guessingProbability + (1 - guessingProbability) / (1 + Math.exp(-(x - inflectionPoint) / slope));
