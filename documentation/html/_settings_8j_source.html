152 // helpers:
153 // if "set == true" the default is set,
154 // otherwise check if outside range or nil, if so set to default
155 + (BOOL) chckBool: (BOOL) val def: (BOOL) def set: (BOOL) set { //console.info("chckBool ", val, "set: ", set);
156  if (!set && !isNaN(val)) return val;
157  return def;
158 }
159 + (int) chckInt: (int) val def: (int) def min: (int) min max: (int) max set: (BOOL) set { //console.info("chckFlt ", val);
160  if (!set && !isNaN(val) && (val <= max) && (val >= min)) return val;
161  return def;
162 }
163 + (int) chckFlt: (float) val def: (float) def min: (float) min max: (float) max set: (BOOL) set { //console.info("chckFlt ", val);
164  if (!set && !isNaN(val) && (val <= max) && (val >= min)) return val;
165  return def;
166 }
