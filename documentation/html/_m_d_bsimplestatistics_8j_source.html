175 function quantileSorted(x, p) {
176  var idx = x.length * p;
177  if (x.length === 0) {
178  throw new Error("quantile requires at least one data point.");
179  } else if (p < 0 || p > 1) {
180  throw new Error("quantiles must be between 0 and 1");
181  } else if (p === 1) {
182  // If p is 1, directly return the last element
183  return x[x.length - 1];
184  } else if (p === 0) {
185  // If p is 0, directly return the first element
186  return x[0];
187  } else if (idx % 1 !== 0) {
188  // If p is not integer, return the next element in array
189  return x[Math.ceil(idx) - 1];
190  } else if (x.length % 2 === 0) {
191  // If the list has even-length, we'll take the average of this number
192  // and the next value, if there is one
193  return (x[idx - 1] + x[idx]) / 2;
194  } else {
195  // Finally, in the simple case of an integer value
196  // with an odd-length list, return the x value at the index.
197  return x[idx];
198  }
199 }

218 function quickselect(arr, k, left, right) {
219  left = left || 0;
220  right = right || arr.length - 1;
221 
222  while (right > left) {
223  // 600 and 0.5 are arbitrary constants chosen in the original paper to minimize execution time
224  if (right - left > 600) {
225  var n = right - left + 1;
226  var m = k - left + 1;
227  var z = Math.log(n);
228  var s = 0.5 * Math.exp((2 * z) / 3);
229  var sd = 0.5 * Math.sqrt((z * s * (n - s)) / n);
230  if (m - n / 2 < 0) { sd *= -1; }
231  var newLeft = Math.max(left, Math.floor(k - (m * s) / n + sd));
232  var newRight = Math.min(
233  right,
234  Math.floor(k + ((n - m) * s) / n + sd)
235  );
236  quickselect(arr, k, newLeft, newRight);
237  }
238 
239  var t = arr[k];
240  var i = left;
241  var j = right;
242 
243  swap(arr, left, k);
244  if (arr[right] > t) { swap(arr, left, right); }
245 
246  while (i < j) {
247  swap(arr, i, j);
248  i++;
249  j--;
250  while (arr[i] < t) { i++; }
251  while (arr[j] > t) { j--; }
252  }
253 
254  if (arr[left] === t) { swap(arr, left, j); }
255  else {
256  j++;
257  swap(arr, j, right);
258  }
259 
260  if (j <= k) { left = j + 1; }
261  if (k <= j) { right = j - 1; }
262  }
263 }

293 function quantile(x, p) {
294  var copy = x.slice();
295 
296  if (Array.isArray(p)) {
297  // rearrange elements so that each element corresponding to a requested
298  // quantile is on a place it would be if the array was fully sorted
299  multiQuantileSelect(copy, p);
300  // Initialize the result array
301  var results = [];
302  // For each requested quantile
303  for (var i = 0; i < p.length; i++) {
304  results[i] = quantileSorted(copy, p[i]);
305  }
306  return results;
307  } else {
308  var idx = quantileIndex(copy.length, p);
309  quantileSelect(copy, idx, 0, copy.length - 1);
310  return quantileSorted(copy, p);
311  }
312 }

324 function multiQuantileSelect(arr, p) {
325  var indices = [0];
326  for (var i = 0; i < p.length; i++) {
327  indices.push(quantileIndex(arr.length, p[i]));
328  }
329  indices.push(arr.length - 1);
330  indices.sort(compare);
331 
332  var stack = [0, indices.length - 1];
333 
334  while (stack.length) {
335  var r = Math.ceil(stack.pop());
336  var l = Math.floor(stack.pop());
337  if (r - l <= 1) { continue; }
338 
339  var m = Math.floor((l + r) / 2);
340  quantileSelect(
341  arr,
342  indices[m],
343  Math.floor(indices[l]),
344  Math.ceil(indices[r])
345  );
346 
347  stack.push(l, m, m, r);
348  }
349 }

355 function quantileIndex(len, p) {
356  var idx = len * p;
357  if (p === 1) {
358  // If p is 1, directly return the last index
359  return len - 1;
360  } else if (p === 0) {
361  // If p is 0, directly return the first index
362  return 0;
363  } else if (idx % 1 !== 0) {
364  // If index is not integer, return the next index in array
365  return Math.ceil(idx) - 1;
366  } else if (len % 2 === 0) {
367  // If the list has even-length, we'll return the middle of two indices
368  // around quantile to indicate that we need an average value of the two
369  return idx - 0.5;
370  } else {
371  // Finally, in the simple case of an integer index
372  // with an odd-length list, return the index
373  return idx;
374  }
375 }
