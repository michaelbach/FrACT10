174function quantileSorted(x, p) {
175 const idx = x.length * p;
176 if (x.length === 0) {
177 throw new Error("quantile requires at least one data point.");
178 } else if (p < 0 || p > 1) {
179 throw new Error("quantiles must be between 0 and 1");
180 } else if (p === 1) {
181 // If p is 1, directly return the last element
182 return x[x.length - 1];
183 } else if (p === 0) {
184 // If p is 0, directly return the first element
185 return x[0];
186 } else if (idx % 1 !== 0) {
187 // If p is not integer, return the next element in array
188 return x[Math.ceil(idx) - 1];
189 } else if (x.length % 2 === 0) {
190 // If the list has even-length, we'll take the average of this number
191 // and the next value, if there is one
192 return (x[idx - 1] + x[idx]) / 2;
193 } else {
194 // Finally, in the simple case of an integer value
195 // with an odd-length list, return the x value at the index.
196 return x[idx];
197 }
198}

217function quickselect(arr, k, left, right) {
218 left = left || 0;
219 right = right || arr.length - 1;
220
221 while (right > left) {
222 // 600 and 0.5 are arbitrary constants chosen in the original paper to minimize execution time
223 if (right - left > 600) {
224 const n = right - left + 1;
225 const m = k - left + 1;
226 const z = Math.log(n);
227 const s = 0.5 * Math.exp((2 * z) / 3);
228 let sd = 0.5 * Math.sqrt((z * s * (n - s)) / n);
229 if (m - n / 2 < 0) { sd *= -1; }
230 const newLeft = Math.max(left, Math.floor(k - (m * s) / n + sd));
231 const newRight = Math.min(
232 right,
233 Math.floor(k + ((n - m) * s) / n + sd)
234 );
235 quickselect(arr, k, newLeft, newRight);
236 }
237
238 const t = arr[k];
239 let i = left;
240 let j = right;
241
242 swap(arr, left, k);
243 if (arr[right] > t) { swap(arr, left, right); }
244
245 while (i < j) {
246 swap(arr, i, j);
247 i++;
248 j--;
249 while (arr[i] < t) { i++; }
250 while (arr[j] > t) { j--; }
251 }
252
253 if (arr[left] === t) { swap(arr, left, j); }
254 else {
255 j++;
256 swap(arr, j, right);
257 }
258
259 if (j <= k) { left = j + 1; }
260 if (k <= j) { right = j - 1; }
261 }
262}

292function quantile(x, p) {
293 const copy = x.slice();
294
295 if (Array.isArray(p)) {
296 // rearrange elements so that each element corresponding to a requested
297 // quantile is on a place it would be if the array was fully sorted
298 multiQuantileSelect(copy, p);
299 // Initialize the result array
300 let results = [];
301 // For each requested quantile
302 for (let i = 0; i < p.length; i++) {
303 results[i] = quantileSorted(copy, p[i]);
304 }
305 return results;
306 } else {
307 const idx = quantileIndex(copy.length, p);
308 quantileSelect(copy, idx, 0, copy.length - 1);
309 return quantileSorted(copy, p);
310 }
311}

313function quantileSelect(arr, k, left, right) {
314 if (k % 1 === 0) {
315 quickselect(arr, k, left, right);
316 } else {
317 k = Math.floor(k);
318 quickselect(arr, k, left, right);
319 quickselect(arr, k + 1, k + 1, right);
320 }

323function multiQuantileSelect(arr, p) {
324 const indices = [0];
325 for (let i = 0; i < p.length; i++) {
326 indices.push(quantileIndex(arr.length, p[i]));
327 }
328 indices.push(arr.length - 1);
329 indices.sort(compare);
330
331 const stack = [0, indices.length - 1];
332
333 while (stack.length) {
334 const r = Math.ceil(stack.pop());
335 const l = Math.floor(stack.pop());
336 if (r - l <= 1) { continue; }
337
338 const m = Math.floor((l + r) / 2);
339 quantileSelect(
340 arr,
341 indices[m],
342 Math.floor(indices[l]),
343 Math.ceil(indices[r])
344 );
345
346 stack.push(l, m, m, r);
347 }
348}

354function quantileIndex(len, p) {
355 const idx = len * p;
356 if (p === 1) {
357 // If p is 1, directly return the last index
358 return len - 1;
359 } else if (p === 0) {
360 // If p is 0, directly return the first index
361 return 0;
362 } else if (idx % 1 !== 0) {
363 // If index is not integer, return the next index in array
364 return Math.ceil(idx) - 1;
365 } else if (len % 2 === 0) {
366 // If the list has even-length, we'll return the middle of two indices
367 // around quantile to indicate that we need an average value of the two
368 return idx - 0.5;
369 } else {
370 // Finally, in the simple case of an integer index
371 // with an odd-length list, return the index
372 return idx;
373 }
374}
