52- (void) drawLineGaussProfileVerticalAtX: (float) x0 y0: (float) y0 y1: (float) y1 sigma: (float) sigma { //console.info("FractControllerAcuityVernier>>DrawLineGaussianProfileVertical ", x0, y0, y1);
53 const ix0 = Math.round(x0);
54 const iSigma = Math.round(Math.max(5, Math.min(sigma * 4, 30))); //trace(sigma, iSigma);
55 CGContextSetLineWidth(cgc, 1);
56 const backGray = [MiscLight upperLuminanceFromContrastMilsn: [MiscLight contrastMichelsonPercentFromWeberPercent: [Settings contrastAcuityWeber]]];
57 const cnt = [Settings contrastAcuityWeber] / 100;
58 for (let ix = ix0 - iSigma; ix <= ix0 + iSigma; ix++) {
59 const gaussValue = Math.exp(-Math.pow(x0 - ix, 2) / sigma);
60 const grayValue = [MiscLight devicegrayFromLuminance: backGray - cnt * gaussValue];
61 CGContextSetStrokeColor(cgc, [CPColor colorWithWhite: grayValue alpha: 1]);
62 CGContextBeginPath(cgc);
63 CGContextMoveToPoint(cgc, ix, y0);
64 CGContextAddLineToPoint(cgc, ix, y1);
65 CGContextStrokePath(cgc);
66 }
